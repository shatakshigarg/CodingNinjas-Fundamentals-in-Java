public class Solution {  

    public static int pairSum(int arr[], int x) {
    	//Your code goes here
       // int x=s.nextInt();
        int n =arr.length;
        int c=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(x==arr[i]+arr[j]){
                    c++;
                   // System.out.println("("+arr[i]+","+arr[j]+")");
                }
            }
        }
        return c;
    }
}
